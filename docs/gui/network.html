Quick Start Guide

Install the Search Query package and start processing queries in minutes: 

# Install SearchQuery
pip install search-query

# Import and use
from search_query import SearchQuery

# Create a query
query = SearchQuery()
query.from_string("(platform*[ti] OR \"digital work\"[ti]) AND 2020-2025[dp]", platform="wos")

# Translate to another platform
translated_query = query.to_string(platform="pubmed")
